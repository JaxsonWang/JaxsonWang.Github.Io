JavaScript图片处理(改)

发表于 2017-05-17 | 更新于 2018-09-29 | 分类于 编程技术

简介

公司做的项目需要，找了以前的代码，发现有些问题，花了点时间把代码优化了下。

公司业务如下：用户上传图片，显示缩略图，传给后台图片数据即可。

所以结合以前做过的思路如下：

设置一个div，这个div负责渲染图片，然后div覆盖file类的input标签，负责上传图片。图片上传采用base64方式，目前也就这个比较流行点。

/*input 1*/
document.getElementById("input_img_1").addEventListener("change",upload_1,false);
function upload_1(e) {
    var files = this.files;
    //console.log(files);
    var img = new Image();
    var reader = new FileReader();
    var fileSize = files[0].size;
    var fileType = files[0].type;
    // 1024字节 = 1千字节 1024千字节 = 1兆字节
    var fileSizeMax = (1024 * 1024) * 5;
    if( (fileType !== "image/jpg") && (fileType !== "image/png") && (fileType !== "image/gif")&&(fileType !== "image/bmp") && (fileType !== "image/jpeg") && (fileType !== "image/gif") ) {
        mui.alert("请上传标准图片格式文件","照片上传");
        return;
    } if (fileSize > fileSizeMax) {
        mui.alert("图片过大，请重新选择","照片上传");
        return;
    } else {
        lrz(files[0]).then(function (rst) {
            //console.log("图片压缩详细输出信息：" + rst);
            //console.log("图片压缩后Base64输出：" + rst.base64);
            reader.readAsDataURL(files[0]);
            reader.onload=function(e){
                /*未压缩*/
                //imgSrc = img.src;
                //console.log("用户上传图片：" + imgSrc);
                img.src=this.result;
                /*已压缩*/
                imgSrc = rst.base64;
                //console.log("用户上传图片（已压缩）：" + imgSrc);
                //console.log("用户上传图片：" + rst.origin.name);
                /*截取字符串*/
                var imgIndex = imgSrc.lastIndexOf(",");
                var imgBase64 = imgSrc.substring(imgIndex + 1);
                //console.log("截取后字符串：" + imgBase64);
                img.style.width = '100%';
                img.style.height = '100%';
                document.getElementById("images_1").innerHTML = "";
                document.getElementById("images_1").appendChild(img);
            }
        });
    }
    console.log("本地上传文件："+files[0].name+"上传成功");
}
//删除图片
var isImage_1 = document.getElementById("images_1");
isImage_1.addEventListener('click',function () { //使用click避免出现在移动端再次点击input的事件
    if (isImage_1.childNodes.length === 1) {
        isImage_1.childNodes[0].remove();
        isImage_1.innerHTML = '<span>上传图片</span><input id="input_img_1" type="file">';
        document.getElementById("input_img_1").addEventListener("change",upload_1,false);
    }
});
