链接数据库实例

<code class="javascript">/**
 * Created by Jaxson on 2017/4/27.
 */
const mysql = require('mysql');
const connection = mysql.createConnection({
    host : 'localhost',
    user : 'root',
    password : '',
    database : 'node'
});
//链接数据库
connection.connect((err,data) => {
    if (err) {
        console.log("数据库链接失败");
    }
});
/**
 * @param {Object} error 回调查询错位信息
 * @param {Object} results 回调查询结果
 * @param {Object} fields 回调字段信息
 */
connection.query('SELECT 1 + 1 AS solution', function (error, results, fields) {
    if (error) throw error;
    console.log('The solution is: ', results[0].solution);
});
connection.end();

<code class="javascript">/**
 * Created by Jaxson on 2017/4/27.
 */
const mysql = require('mysql');
const connection = mysql.createConnection({
    host : 'localhost',
    user : 'root',
    password : '',
    database : 'node'
});
//链接数据库
connection.connect((err,data) => {
    if (err) {
        console.log("数据库链接失败");
    }
});
const addSql = 'INSERT INTO websites(Id,name,url,alexa,country) VALUES(0,?,?,?,?)';
const addSqlParams = ['淮城一只猫', 'https://iiong.com','100000000', 'CN'];
/**
 * @param {Object} error 回调查询错位信息
 * @param {Object} results 回调查询结果
 * @param {Object} fields 回调字段信息
 */
connection.query(addSql,addSqlParams,(error,results,fields) => {
    if (error) {
        console.error(err);
    }
    console.log('====================');
    console.log('INSERT ID:',results);
    console.log('====================');
});
connection.end();

改

<code class="javascript">/**
 * Created by Jaxson on 2017/4/27.
 */
const mysql = require('mysql');
const connection = mysql.createConnection({
    host : 'localhost',
    user : 'root',
    password : '',
    database : 'node'
});
//链接数据库
connection.connect((err,data) => {
    if (err) {
        console.log("数据库链接失败");
    }
});
const updateSql = 'UPDATE websites SET name = ?,url = ? WHERE Id = ?';
const updateSqlParams = ['王瑾的博客', 'https://boxrom.com',6];
/**
 * @param {Object} error 回调查询错位信息
 * @param {Object} results 回调查询结果
 * @param {Object} fields 回调字段信息
 */
connection.query(updateSql,updateSqlParams,(error,results,fields) => {
    if (error) {
        console.error(err);
    }
    console.log('====================');
    console.log('UPDATE affectedRows',results.affectedRows);
    console.log('====================');
});
connection.end();
